2.2 app/__init__.py
python
from flask import Flask
from flask_sqlalchemy import SQLAlchemy
from config import Config

db = SQLAlchemy()

def create_app(config_class=Config):
    app = Flask(__name__, instance_relative_config=True)
    app.config.from_object(config_class)

    db.init_app(app)

    # 确保 instance 文件夹存在
    import os
    try:
        os.makedirs(app.instance_path)
    except OSError:
        pass

    # 导入模型并创建数据库表
    from app.models import Question
    with app.app_context():
        db.create_all()

    # 注册蓝图路由
    from app.routes import bp as routes_bp
    app.register_blueprint(routes_bp)

    return app

from app import db

class Question(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    text = db.Column(db.Text, nullable=False)
    answer = db.Column(db.String(200), nullable=False)
    explanation = db.Column(db.Text, nullable=True)
    difficulty = db.Column(db.String(20), default='medium')  # easy, medium, hard
    created_at = db.Column(db.DateTime, default=db.func.current_timestamp())

    def __repr__(self):
        return f'<Question {self.text[:20]}>'
2.4 app/routes.py
python
from flask import Blueprint, render_template, redirect, url_for, flash, request
from app import db
from app.models import Question
from flask_wtf import FlaskForm
from wtforms import StringField, TextAreaField, SelectField, SubmitField
from wtforms.validators import DataRequired, Length

bp = Blueprint('main', __name__)

class QuestionForm(FlaskForm):
    text = TextAreaField('题目内容', validators=[DataRequired(), Length(min=5)])
    answer = StringField('答案', validators=[DataRequired(), Length(max=200)])
    explanation = TextAreaField('解析', validators=[Length(max=500)])
    difficulty = SelectField('难度', choices=[('easy', '简单'), ('medium', '中等'), ('hard', '困难')], default='medium')
    submit = SubmitField('提交')

@bp.route('/')
def index():
    questions = Question.query.order_by(Question.created_at.desc()).all()
    return render_template('index.html', questions=questions)

@bp.route('/question/<int:id>')
def view_question(id):
    q = Question.query.get_or_404(id)
    return render_template('view_question.html', question=q)

@bp.route('/question/new', methods=['GET', 'POST'])
def new_question():
    form = QuestionForm()
    if form.validate_on_submit():
        q = Question(
            text=form.text.data,
            answer=form.answer.data,
            explanation=form.explanation.data,
            difficulty=form.difficulty.data
        )
        db.session.add(q)
        db.session.commit()
        flash('题目已添加', 'success')
        return redirect(url_for('main.index'))
    return render_template('question_form.html', form=form, title='添加题目')

@bp.route('/question/<int:id>/edit', methods=['GET', 'POST'])
def edit_question(id):
    q = Question.query.get_or_404(id)
    form = QuestionForm(obj=q)
    if form.validate_on_submit():
        q.text = form.text.data
        q.answer = form.answer.data
        q.explanation = form.explanation.data
        q.difficulty = form.difficulty.data
        db.session.commit()
        flash('题目已更新', 'success')
        return redirect(url_for('main.view_question', id=q.id))
    return render_template('question_form.html', form=form, title='编辑题目')

@bp.route('/question/<int:id>/delete', methods=['POST'])
def delete_question(id):
    q = Question.query.get_or_404(id)
    db.session.delete(q)
    db.session.commit()
    flash('题目已删除', 'success')
    return redirect(url_for('main.index'))

4. Gunicorn 配置
生产环境使用 Gunicorn 作为 WSGI 服务器，监听本地端口（如 127.0.0.1:8000），由 Nginx 反向代理。

4.1 手动启动测试
bash
cd /home/your_user/questions_bank
source venv/bin/activate
gunicorn --workers 3 --bind 127.0.0.1:8000 run:app
此时可通过 curl http://127.0.0.1:8000 测试（应返回 HTML 内容）。

4.2 配置 Systemd 服务（见第 6 节）实现后台运行及开机自启。
5. Nginx 反向代理配置
5.1 禁用默认站点
bash
sudo rm /etc/nginx/sites-enabled/default   # 删除默认站点链接
5.2 创建题库站点配置文件
新建 /etc/nginx/sites-available/questions_bank：

nginx
server {
    listen 80;
    server_name _;   # 或者写你的服务器 IP / 域名

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location /static/ {
        alias /home/your_user/questions_bank/app/static/;
        expires 30d;
    }
}
5.3 启用配置
bash
sudo ln -s /etc/nginx/sites-available/questions_bank /etc/nginx/sites-enabled/
sudo nginx -t          # 测试配置语法
sudo systemctl restart nginx
6. Systemd 服务配置（开机自启）
创建服务文件 /etc/systemd/system/questions_bank.service：

ini
[Unit]
Description=Gunicorn instance for questions_bank
After=network.target

[Service]
User=your_user
Group=www-data
WorkingDirectory=/home/your_user/questions_bank
Environment="PATH=/home/your_user/questions_bank/venv/bin"
ExecStart=/home/your_user/questions_bank/venv/bin/gunicorn --workers 3 --bind 127.0.0.1:8000 run:app

[Install]
WantedBy=multi-user.target
注意：将 your_user 替换为实际用户名，路径根据实际情况修改。

启动并启用服务：

bash
sudo systemctl start questions_bank
sudo systemctl enable questions_bank
sudo systemctl status questions_bank   # 查看运行状态
7. 防火墙设置
使用 ufw 开放 HTTP（80）端口：

bash
sudo ufw allow 'Nginx HTTP'
# 或者直接允许 80 端口
sudo ufw allow 80/tcp
如果之后启用 HTTPS，还需开放 443 端口：

bash
sudo ufw allow 'Nginx HTTPS'
检查防火墙状态：

bash
sudo ufw status

9. 常见问题排查
现象	可能原因	解决方法
访问 IP 显示 Nginx 默认页面	默认站点未禁用	sudo rm /etc/nginx/sites-enabled/default
出现 502 Bad Gateway	Gunicorn 未运行或端口/地址不一致	sudo systemctl status questions_bank；检查 proxy_pass 与 Gunicorn --bind 是否一致
出现 403 Forbidden	Nginx 权限不足	检查 location / 的 proxy_pass 配置；确保用户 www-data 可访问项目目录
出现 404 Not Found	路由未匹配或 Gunicorn 未正确启动	查看 Gunicorn 日志：sudo journalctl -u questions_bank
静态文件 404	Nginx 中 alias 路径错误	确认 alias 指向 app/static/ 的绝对路径
无法连接（连接超时）	防火墙未开放 80 端口	sudo ufw allow 'Nginx HTTP'
数据库表未创建	instance/ 目录无写入权限	chmod 755 instance/ 或手动执行 db.create_all()
常用调试命令
查看 Gunicorn 日志：sudo journalctl -u questions_bank -f

查看 Nginx 错误日志：sudo tail -f /var/log/nginx/error.log

测试 Nginx 配置：sudo nginx -t

重启服务：sudo systemctl restart questions_bank 或 sudo systemctl restart nginx
